170  double logOfProduct(const IndexSet& s) const {
171  if (s.last() >= numPrimes())
172  Error("FHEContext::logOfProduct: IndexSet has too many rows");
173 
174  double ans = 0.0;
175  for (long i = s.first(); i <= s.last(); i = s.next(i))
176  ans += logOfPrime(i);
177  return ans;
178  }
