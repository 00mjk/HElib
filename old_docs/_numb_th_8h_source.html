644 // Example: rotate by 1 means [0 1 2 3] -> [3 0 1 2]
645 // rotate by -1 means [0 1 2 3] -> [1 2 3 0]
646 template<class T> void rotate(Vec<T>& v, long k)
647 {
648  long n = v.length();
649  if (n <= 1) return;
650 
651  k %= n;
652  if (k < 0) k += n;
653 
654  if (k == 0) return;
655 
656  reverse(v, 0, n-1);
657  reverse(v, 0, k-1);
658  reverse(v, k, n-1);
659 }

802 // plaintextAutomorph: Compute b(X) = a(X^k) mod Phi_m(X).
803 // Result is calclated in the output b "in place", so a should not alias b.
804 template <class RX, class RXModulus>
805 void plaintextAutomorph(RX& b, const RX& a, long k, long m, const RXModulus& PhimX)
806 {
807  // compute b(X) = a(X^k) mod (X^m-1)
808  b.SetLength(m);
809  for (long j = 0; j < m; j++) b[j] = 0;
810  mulmod_precon_t precon = PrepMulModPrecon(k, m);
811  for (long j = 0; j <= deg(a); j++)
812  b[MulModPrecon(j, k, m, precon)] = a[j]; // b[j*k mod m] = a[j]
813  b.normalize();
814 
815  rem(b, b, PhimX); // reduce modulo the m'th cyclotomic
816 }
