529  template<class T>
530  void genericSelect(Ctxt& ctxt1, const Ctxt& ctxt2,
531  const T& selector) const
532  {
533  if (&ctxt1 == &ctxt2) return; // nothing to do
534 
535  assert(&context == &ctxt1.getContext() && &context == &ctxt2.getContext());
536  ZZX poly;
537  encode(poly,selector); // encode as polynomial
538  DoubleCRT dcrt(poly, context, ctxt1.getPrimeSet());// convert to DoubleCRT
539 
540  ctxt1.multByConstant(dcrt); // keep only the slots with 1's
541 
542  dcrt -= 1; // convert 1 => 0, 0 => -1
543  Ctxt tmp=ctxt2; // a copy of ctxt2
544  tmp.multByConstant(dcrt);// keep (but negate) only the slots with 0's
545  ctxt1 -= tmp;
546  }
