153  return k;
154  }

159  assert(i >= 0 && i < 2*k-1);

309 template <class T>
310 long FullBinaryTree<T>::addChildren(long prntIdx,
311  const T& leftData, const T& rightData)
312 {
313  assert (prntIdx >= 0 && prntIdx < (long)(nodes.size()));
314 
315  // If parent is a leaf, add to it two children
316  if (nodes[prntIdx].leftChild==-1 && nodes[prntIdx].rightChild==-1) {
317  long childIdx = nodes.size();
318  TreeNode<T> n1(leftData);
319  nodes.push_back(n1); // add left child to vector
320  TreeNode<T> n2(rightData);
321  nodes.push_back(n2);// add right child to vector
322 
323  TreeNode<T>& parent = nodes[prntIdx];
324  TreeNode<T>& left = nodes[childIdx];
325  TreeNode<T>& right = nodes[childIdx+1];
326 
327  parent.leftChild = childIdx; // point to children from parent
328  parent.rightChild= childIdx+1;
329  left.parent = right.parent = prntIdx; // point to parent from children
330 
331  // remove parent and insert children to the linked list of leaves
332  left.prev = parent.prev;
333  left.next = childIdx+1;
334  right.prev = childIdx;
335  right.next = parent.next;
336  if (parent.prev>=0) { // parent was not the 1st leaf
337  nodes[parent.prev].next = childIdx;
338  parent.prev = -1;
339  }
340  else // parent was the first leaf, now its left child is 1st
341  frstLeaf = childIdx;
342 
343  if (parent.next>=0) { // parent was not the last leaf
344  nodes[parent.next].prev = childIdx+1;
345  parent.next = -1;
346  }
347  else // parent was the last leaf, now its left child is last
348  lstLeaf = childIdx+1;
349 
350  nLeaves++; // we replaced a leaf by a parent w/ two leaves
351  }
352  else { // parent is not a leaf, update the two children
353  TreeNode<T>& parent = nodes[prntIdx];
354  assert(parent.leftChild>=0 && parent.rightChild>=0);
355 
356  TreeNode<T>& left = nodes[parent.leftChild];
357  TreeNode<T>& right = nodes[parent.rightChild];
358  left.data = leftData;
359  right.data = rightData;
360  }
361  return nodes[prntIdx].leftChild;
362 }
