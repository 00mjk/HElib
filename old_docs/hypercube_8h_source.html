33  void initSignature(const long _dims[], long _ndims)
34  {
35  assert(dims.length() == 0); // can only initialize a NULL signature

37 
38  dims.SetLength(_ndims);
39  prods.SetLength(_ndims+1);
40  prods[_ndims] = 1;

111  template <typename VecType> bool incrementCoords(VecType& v) const {
112  for (long i=getNumDims()-1; i>=0; --i) {
113  if (i>=lsize(v)) continue; // sanity check
114 
115  // increment current index, set all the ones after it to zero
116  if (long(v[i]) < getDim(i)-1) {
117  v[i]++;
118  for (long j=i+1; j<lsize(v); j++) v[j] = 0;
119  return true; // succeeded in incrementing the vector
120  }
121  // if buffer[i] >= getDim(i)-1, move to previous index i
122  }
123  return false; // cannot increment the vector anymore
124  }
